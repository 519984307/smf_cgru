Houdini Afanasy Setup

Afanasy houdini operator libraries located in afanasy/plugins/houdini/houdini[version] directory. You can add it automatically by adding this directory to HOUDINI_OTLSCAN_PATH variable.

Houdini setup example: 


   #!/bin/bash

   # Setup Afanasy if was not
   if [ -z $AF_ROOT ]; then
      pwd=$PWD
      cd ../..
      source setup.sh
      cd $pwd
   fi

   # directory where Houdini installed (default value):
   export HOUDINI_LOCATION=/cg/soft/houdini-10

   # to launch commands from current directory:
   export AF_CMD_PREFIX="./"

   # overrides (set custom values there):
   [ -f override.sh ] && source override.sh

   pwd=$PWD
   cd $HOUDINI_LOCATION
   source houdini_setup_bash
   cd $pwd

   # Set Afanasy houdini scripts version
   export HOUDINI_CGRU_VERSION="10"
   # Set Afanasy houdini scripts location
   export HOUDINI_CGRU_PATH=$AF_ROOT/plugins/houdini/houdini$HOUDINI_CGRU_VERSION

   export HOUDINI_CGRU_OTLSCAN_PATH=$HIH/otls:$HOUDINI_CGRU_PATH/otls:$HH/otls

   if [ "$HOUDINI_OTLSCAN_PATH" != "" ]; then
      export HOUDINI_OTLSCAN_PATH="${HOUDINI_CGRU_OTLSCAN_PATH}:${HOUDINI_OTLSCAN_PATH}"   
   else
      export HOUDINI_OTLSCAN_PATH=$HOUDINI_CGRU_OTLSCAN_PATH
   fi
